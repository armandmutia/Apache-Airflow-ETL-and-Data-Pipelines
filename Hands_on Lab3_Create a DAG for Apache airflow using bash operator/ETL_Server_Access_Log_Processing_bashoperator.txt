# ETL_Server_Access_Log_Processing.py
# using ai tool

# 1. Imports block
from datetime import timedelta
from airflow import DAG
from airflow.operators.bash_operator import BashOperator
from airflow.utils.dates import days_ago

# 2. DAG Arguments block
default_args = {
    'owner': 'Armand Mutia',
    'start_date': days_ago(0),
    'email': ['armand.mutia@gmail.com'],
    'email_on_failure': True,
    'email_on_retry': True,
    'retries': 1,
    'retry_delay': timedelta(minutes=5),
}

# 3. DAG Definition block — runs daily
dag = DAG(
    dag_id='ETL_Server_Access_Log_Processing',
    default_args=default_args,
    description='ETL DAG for server access log',
    schedule_interval=timedelta(days=1),
)

# 4. Download task
download = BashOperator(
    task_id='download',
    bash_command=(
        'wget "https://cf-courses-data.s3.us.cloud-object-storage.appdomain.cloud/'
        'IBM-DB0250EN-SkillsNetwork/labs/Apache%20Airflow/Build%20a%20DAG%20using%20Airflow/'
        'web-server-access-log.txt" '
        '-O /home/project/airflow/dags/web-server-access-log.txt'
    ),
    dag=dag,
)

# 5. Extract task — extracts fields: timestamp (col 1) and visitorid (col 4)
extract = BashOperator(
    task_id='extract',
    bash_command=(
        'cut -d"#" -f1,4 /home/project/airflow/dags/web-server-access-log.txt '
        '> /home/project/airflow/dags/extracted.txt'
    ),
    dag=dag,
)

# 6. Transform task — capitalizes visitorid (field 2 of extracted.txt)
transform = BashOperator(
    task_id='transform',
    bash_command=(
        'tr "[a-z]" "[A-Z]" < /home/project/airflow/dags/extracted.txt '
        '> /home/project/airflow/dags/capitalized.txt'
    ),
    dag=dag,
)

# 7. Load task — compresses the transformed data
load = BashOperator(
    task_id='load',
    bash_command=(
        'zip /home/project/airflow/dags/web-server-access-log.zip '
        '/home/project/airflow/dags/capitalized.txt'
    ),
    dag=dag,
)

# 8. Task pipeline
download >> extract >> transform >> load
